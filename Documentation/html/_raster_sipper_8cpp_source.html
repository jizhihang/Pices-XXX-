571 // Creates a raster from a compressedBuff created by 'SimpleCompression'
572 RasterSipperPtr RasterSipper::FromSimpleCompression (const uchar* compressedBuff,
573  uint32 compressedBuffLen
574  )
575 {
576  // I expect simple run length compressed data to be passed in. The data was orginally
577  // compressed by SimpleCompressor and we will use the same class to decompress.
578 
579  // The format of the uncompressed data is very simple.
580  // Bytes Description
581  // 0 thru 1: Height 2 byte integer.
582  // 2 thru 3: Width 2 byte integer.
583  //
584  // 4 thru (Hight * Width) - 1: Raster Data
585  // Row 0 : (consists of 'Width' bytes.)
586  // Row 1 : ( "" "" "" "" )
587  // -- --- --- ----
588  // Row Height - 1 : (Last row of image)
589  //
590  // Each byte inthe raster represents one pixel 0 - 255 grayscale; where 0=background.
591 
592  uint32 unCompressedSize = 0;
593 
594  if (compressedBuff == NULL)
595  {
596  cerr << endl
597  << "RasterSipper::FromSimpleCompression ***ERROR*** (compressedBuff==NULL)" << endl
598  << endl;
599  return new RasterSipper (20, 20, false);
600  }
601 
602  uchar* uncompressedBuff = SimpleCompressor::Decompress (compressedBuff, compressedBuffLen, unCompressedSize);
603  if (uncompressedBuff == NULL)
604  {
605  cerr << endl
606  << "RasterSipper::FromSimpleCompression ***ERROR*** (uncompressedBuff==NULL)" << endl
607  << endl;
608  return new RasterSipper (20, 20, false);
609  }
610 
611  // The first Four Bytes have the image Height, and Width.
612 
613  int32 height = uncompressedBuff[0] * 256 + uncompressedBuff[1];
614  int32 width = uncompressedBuff[2] * 256 + uncompressedBuff[3];
615 
616  if ((height < 1) || (height > 1000) || (width < 1) || (width > 1000))
617  {
618  cerr << endl
619  << "RasterSipper::FromSimpleCompression ***ERROR*** Height[" << height << "] Width[" << width << "] are out of bounds." << endl
620  << endl;
621  return new RasterSipper (20, 20, false);
622  }
623 
624  uint32 totalPixels = height * width;
625  if (unCompressedSize > (totalPixels + 4))
626  {
627  cerr << endl
628  << "RasterSipper::FromSimpleCompression ***ERROR*** unCompressedSize[" << unCompressedSize << "] is greater than 'totalPixels + 4'[" << totalPixels + 4 << "]." << endl
629  << endl;
630  }
631 
632  RasterSipperPtr result = new RasterSipper (height, width, false);
633  uchar* greenArea = result->GreenArea ();
634 
635  uint32 nextIdx = 4;
636  uint32 greanAreaIdx = 0;
637 
638  while ((nextIdx < (uint32)unCompressedSize) && (greanAreaIdx < totalPixels))
639  {
640  greenArea[greanAreaIdx] = uncompressedBuff[nextIdx];
641  nextIdx++;
642  greanAreaIdx++;
643  }
644 
645  delete uncompressedBuff;
646  uncompressedBuff = NULL;
647 
648  return result;
649 } /* FromSimpleCompression */

654 RasterSipperPtr RasterSipper::FromCompressor (const uchar* compressedBuff, // Creates a raster from a compressedBuff created by 'Compressor'(zlib)
655  uint32 compressedBuffLen
656  )
657 {
658  uint32 unCompressedBuffLen = 0;
659  uchar* unCompressedBuff = NULL;
660  try
661  {
662  unCompressedBuff = (uchar*)Compressor::Decompress (compressedBuff, compressedBuffLen, unCompressedBuffLen);
663  }
664  catch (...)
665  {
666  KKStr errMsg = "RasterSipper::FromCompressor Exception thrown calling 'Compressor::Decompress'.";
667  cerr << errMsg << endl;
668  throw KKStrException (errMsg);
669  }
670 
671  if (!unCompressedBuff)
672  return NULL;
673 
674  if (unCompressedBuffLen < 10)
675  {
676  cerr << std::endl << std::endl << "Raster::FromCompressor Compressor did not return any data." << std::endl;
677  return NULL;
678  }
679 
680  // 0 - 3: Height: high order to low order
681  // 4 - 7: Width: high order to low order
682  // 8 - 8: Color 0 = Grayscale, 1 = Color
683  // 9 - 8 + (Height * Width) Green Channel
684  // xxxxx Red Channel
685  // xxxxx Blue Channel
686 
687  uint32 height = 0;
688  uint32 width = 0;
689  height = unCompressedBuff[0] + unCompressedBuff[1] * 256 + unCompressedBuff[2] * 256 * 256 + unCompressedBuff[3] * 256 * 256 * 256;
690  width = unCompressedBuff[4] + unCompressedBuff[5] * 256 + unCompressedBuff[6] * 256 * 256 + unCompressedBuff[7] * 256 * 256 * 256;
691 
692  bool color = (unCompressedBuff[8] == 1);
693 
694  uint32 totalPixels = height * width;
695  if ((totalPixels > (100 * 1024 * 1024)) || (height < 1) || (width < 1))
696  {
697  cerr << std::endl << std::endl << "Raster::FromCompressor Height[" << height << "] Width[" << width << "] is not valid." << std::endl << std::endl;
698  delete unCompressedBuff;
699  unCompressedBuff = NULL;
700  return NULL;
701  }
702 
703  uint32 totalDataNeeded = totalPixels + (color ? (2 * totalPixels) : 0) + 9;
704  if (totalDataNeeded > unCompressedBuffLen)
705  {
706  cerr << std::endl << std::endl
707  << "Raster::FromCompressor Height[" << height << "] Width[" << width << "] Color[" << (color ? "Yes" : "No") << "]" << std::endl
708  << " requires TotalDataNeeded[" << totalDataNeeded << "] but UnCompressedBuffLen[" << unCompressedBuffLen << "]" << std::endl
709  << std::endl;
710  delete unCompressedBuff;
711  unCompressedBuff = NULL;
712  return NULL;
713  }
714 
715  RasterSipperPtr r = new RasterSipper ((int32)height, (int32)width, color);
716 
717  uint32 nextIdx = 9;
718  uint32 x;
719 
720  uchar* greenArea = r->GreenArea ();
721  for (x = 0; x < totalPixels; x++, nextIdx++)
722  greenArea[x] = unCompressedBuff[nextIdx];
723 
724  if (color)
725  {
726  uchar* redArea = r->RedArea ();
727  for (x = 0; x < totalPixels; x++, nextIdx++)
728  redArea[x] = unCompressedBuff[nextIdx];
729 
730  uchar* blueArea = r->BlueArea ();
731  for (x = 0; x < totalPixels; x++, nextIdx++)
732  blueArea[x] = unCompressedBuff[nextIdx];
733  }
734 
735  delete unCompressedBuff;
736  unCompressedBuff = NULL;
737 
738  return r;
739 } /* FromCompressor */
