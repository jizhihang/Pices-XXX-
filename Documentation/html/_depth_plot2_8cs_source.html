803  private void ChangeDataFieldLabel (int dataLabelIdx)
804  {
805  if ((dataLabelIdx < 0) || (dataLabelIdx >= InstrumentData.NumOfFields ()))
806  {
807  MessageBox.Show ("'ChangeDataFieldLabel' Invalid dataLabelIdx[" + dataLabelIdx.ToString () + "].");
808  return;
809  }
810 
811  int dataFieldIdx = dataLabels[dataLabelIdx].SelectedIndex;
812 
813  if ((dataFieldIdx < 0) || (dataFieldIdx >= InstrumentData.NumDataFields ()))
814  {
815  MessageBox.Show ("'ChangeDataFieldLabel' Invalid dataFieldIdx[" + dataFieldIdx.ToString () + "].");
816  return;
817  }
818 
819 
820  if (fieldIdxs[plottingAxisIdx, dataLabelIdx] == dataFieldIdx)
821  {
822  // Still pointiing to same label, there is nothing to do.
823  return;
824  }
825 
826  int x;
827 
828  // First see if another dataLabel is already pointing to this data field.
829  for (x = 0; x < InstrumentData.NumOfFields(); x++)
830  {
831  if (fieldIdxs[plottingAxisIdx, x] == dataFieldIdx)
832  {
833  // Another label is already referencing the data label; so we will change
834  // its assignment to the one that 'dataLabelIdx' is referencing.
835  fieldIdxs[plottingAxisIdx, x] = fieldIdxs[plottingAxisIdx, dataLabelIdx];
836  if (x < dataLabels.Length)
837  dataLabels[x].SelectedIndex = fieldIdxs[plottingAxisIdx, x];
838  break;
839  }
840  }
841 
842  fieldIdxs[plottingAxisIdx, dataLabelIdx] = dataFieldIdx;
843  dataLabels[dataLabelIdx].SelectedIndex = dataFieldIdx;
844 
845  PopulateDataFieldLabels ();
846  PopulateDataFields (history.CurrentDataRow ());
847  PaintChart ();
848  } /* ChangeDataFieldLabel */

925  void LoadConfiguration ()
926  {
927  StreamReader f = null;
928 
929  try
930  {
931  f = new StreamReader (dataWindowConfigFileName);
932  }
933  catch (Exception)
934  {
935  return;
936  }
937 
938  int[,] newFieldIdxs = new int [2, InstrumentData.NumOfFields ()];
939  bool[,] fieldsFound = new bool [2, InstrumentData.NumOfFields ()];
940  int z;
941 
942  for (z = 0; z < InstrumentData.NumOfFields (); z++)
943  {
944  newFieldIdxs[0, z] = 0;
945  newFieldIdxs[1, z] = 0;
946  fieldsFound [0, z] = false;
947  fieldsFound [1, z] = false;
948  }
949 
950  int numFieldsFound0 = 0;
951  int numFieldsFound1 = 0;
952 
953 
954  string txtLine = f.ReadLine ();
955  while (txtLine != null)
956  {
957  if (txtLine.Substring (0, 2) != "//")
958  {
959  String[] fields = txtLine.Split ('\t');
960  if (fields.Length > 1)
961  {
962  string rowType = fields[0].ToLower ().Trim ();
963 
964  if (rowType== "fieldidx")
965  {
966  int idx0 = InstrumentData.GetFieldNameIndex (fields[1]);
967  int idx1 = idx0;
968  if (fields.Length > 2)
969  idx1 = InstrumentData.GetFieldNameIndex (fields[2]);
970 
971  if (idx0 >= 0)
972  {
973  // Make sure we have not already seen this idx already.
974 
975  if (!fieldsFound[0, idx0])
976  {
977  newFieldIdxs[0, numFieldsFound0] = idx0;
978  fieldsFound[0, idx0] = true;
979  numFieldsFound0++;
980  }
981  }
982 
983  if (idx1 >= 0)
984  {
985  // Make sure we have not already seen this idx already.
986 
987  if (!fieldsFound[1, idx1])
988  {
989  newFieldIdxs[1, numFieldsFound1] = idx1;
990  fieldsFound[1, idx1] = true;
991  numFieldsFound1++;
992  }
993  }
994  }
995  else if (rowType == "minmax")
996  {
997  if (fields.Length < 4)
998  continue;
999 
1000  int idx = InstrumentData.GetFieldNameIndex (fields[1]);
1001  if ((idx >= 0) && (idx < minValues.Length))
1002  {
1003  minValues[idx] = StrToFloat (fields[2]);
1004  maxValues[idx] = StrToFloat (fields[3]);
1005  if (fields.Length < 5)
1006  {
1007  plotData [idx] = true;
1008  }
1009  else
1010  {
1011  plotData [idx] = PicesKKStr.StrToBool (fields[4]);
1012  }
1013  }
1014  }
1015 
1016  else if (rowType == "plottingaxis")
1017  {
1018  if (fields.Length == 2)
1019  {
1020  plottingAxisEventsAreSuspended = true;
1021  UpdatePlotingAxis ();
1022  PopulateDataFieldLabels ();
1023 
1024  plottingAxisEventsAreSuspended = false;
1025  }
1026  }
1027  }
1028  }
1029 
1030  txtLine = f.ReadLine();
1031  }
1032 
1033  f.Close();
1034 
1035  // Lets account for missing fields. This should never be able to happen unless a
1036  // another entity/user modifies the configuration file.
1037  for (z = 0; z < InstrumentData.NumOfFields(); z++)
1038  {
1039  if (!fieldsFound[0, z])
1040  {
1041  newFieldIdxs[0, numFieldsFound0] = z;
1042  numFieldsFound0++;
1043  }
1044  }
1045 
1046  for (z = 0; z < InstrumentData.NumOfFields(); z++)
1047  {
1048  if (!fieldsFound[1, z])
1049  {
1050  newFieldIdxs[1, numFieldsFound1] = z;
1051  numFieldsFound1++;
1052  }
1053  }
1054 
1055  fieldIdxs = newFieldIdxs;
1056 
1057  for (z = 0; z < dataLabels.Length; z++)
1058  dataLabels[z].SelectedIndex = fieldIdxs[plottingAxisIdx, z];
1059 
1060  chartParms.SetFieldIdxs (fieldIdxs);
1061 
1062  PopulateDataFieldLabels ();
1063  SetChartRanges ();
1064 
1065  return;
1066  } /* LoadConfiguration */
