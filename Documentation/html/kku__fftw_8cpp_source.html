144 //number_of_complex_samples -> number of samples (N^2 order number)
145 //isign -> 1 to calculate FFT and -1 to calculate Reverse FFT
146 void KKU::FFT (float data[],
147  uint32 number_of_complex_samples,
148  int32 isign
149  )
150 {
151  //variables for trigonometric recurrences
152  uint32 n = 0;
153  uint32 mmax = 0;
154  uint32 m = 0;
155  uint32 j = 0;
156  uint32 istep = 0;
157  uint32 i = 0;
158 
159  float wtemp = 0.0;
160  float wr = 0.0;
161  float wpr = 0.0;
162  float wpi = 0.0;
163  float wi = 0.0;
164  float theta = 0.0;
165  float tempr = 0.0;
166  float tempi = 0.0;
167 
168  /*
169  * The Bit-Reversal Method
170  *
171  * First, the original array must be transformed in order to perform the Danielson-Lanzcos
172  * algorithm. For example, the complex[index] will swap places with the
173  * complex[bit-reverse-order-index]. If the index (in binary) is 0b00001, the
174  * bit-reverse-order-index will be 0b10000. In figure 1, you can see what happens to the
175  * data array after the transformation.
176  *
177  * The implementation of this method according to Numerical Recipes In C goes like this:
178  */
179 
180  //the complex array is real+complex so the array
181  //as a size n = 2* number of complex samples
182  // real part is the data[index] and
183  //the complex part is the data[index+1]
184  n = number_of_complex_samples * 2;
185 
186  //binary inversion (note that the indexes
187  //start from 0 witch means that the
188  //real part of the complex is on the even-indexes
189  //and the complex part is on the odd-indexes
190  j = 0;
191  for (i = 0; i < n / 2; i += 2)
192  {
193  if (j > i)
194  {
195  //swap the real part
196  SWAP (data[j], data[i]);
197 
198  //swap the complex part
199  SWAP (data[j + 1],data[i + 1]);
200 
201  // checks if the changes occurs in the first half
202  // and use the mirrored effect on the second half
203  if ((j / 2) < (n / 4))
204  {
205  //swap the real part
206  SWAP (data[(n - (i + 2))], data[(n - (j + 2))]);
207  //swap the complex part
208  SWAP (data[(n - (i + 2)) + 1], data[(n - (j + 2)) + 1]);
209  }
210  }
211  m = n / 2;
212  while (m >= 2 && j >= m)
213  {
214  j -= m;
215  m = m / 2;
216  }
217  j += m;
218  }
219 
220  //Danielson-Lanzcos routine
221  mmax = 2;
222  //external loop
223  while (n > mmax)
224  {
225  istep = mmax << 1;
226  theta = isign * (2 * (float)PIE / mmax);
227  wtemp = sin (0.5f * theta);
228  wpr = -2.0f * wtemp * wtemp;
229  wpi = sin (theta);
230  wr = 1.0f;
231  wi = 0.0f;
232  //internal loops
233  for (m = 1; m < mmax; m += 2)
234  {
235  for (i = m; i <= n; i += istep)
236  {
237  j = i + mmax;
238  tempr = wr * data[j - 1] - wi * data[j];
239  tempi = wr * data[j] + wi * data[j - 1];
240  data[j - 1] = data[i - 1] - tempr;
241  data[j] = data[i] - tempi;
242  data[i - 1] += tempr;
243  data[i] += tempi;
244  }
245  wr = (wtemp = wr) * wpr - wi * wpi + wr;
246  wi = wi * wpr + wtemp * wpi + wi;
247  }
248  mmax = istep;
249  }
250 } /* FFT */
