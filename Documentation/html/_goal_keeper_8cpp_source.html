105 void GoalKeeper::StartBlock ()
106 {
107  int32 curThreadId = KKU::osGetThreadId ();
108 
109  bool firstPassThruLoop = true;
110  bool weAreBlocked = true;
111  unsigned long milliSecsBlocked = 0;
112 
113  while (weAreBlocked)
114  {
115  CriticalSectionStart ();
116 
117  if (firstPassThruLoop)
118  numBlockedThreads++;
119 
120  if (blocked)
121  {
122  if (curThreadId == blockerThreadId)
123  {
124  // We are the thread that already holds the block; so ok for us
125  // to procees.
126  blockerDepth++;
127  weAreBlocked = false;
128  numBlockedThreads--;
129  }
130  else
131  {
132  weAreBlocked = true;
133  if (firstPassThruLoop)
134  numOfBlockingInstances++;
135  }
136  }
137  else
138  {
139  // No one holds the lock; so we can take it.
140  blocked = true;
141  blockerDepth = 1;
142  blockerThreadId = curThreadId;
143  weAreBlocked = false;
144  numBlockedThreads--;
145  }
146 
147  CriticalSectionEnd ();
148 
149  if (weAreBlocked)
150  {
151  if (milliSecsBlocked < 10)
152  {
153  osSleep (0.001f);
154  milliSecsBlocked++;
155  }
156 
157  else if (milliSecsBlocked < 200)
158  {
159  osSleep (0.010f);
160  milliSecsBlocked += 10;
161  }
162 
163  else if (milliSecsBlocked < 10000)
164  {
165  osSleep (0.100f);
166  milliSecsBlocked += 100;
167  }
168 
169  else
170  {
171  osSleep (0.400f);
172  milliSecsBlocked += 400;
173  }
174  }
175 
176  firstPassThruLoop = false;
177  }
178 
179  return;
180 } /* StartBlock */

185 void GoalKeeper::EndBlock ()
186 {
187  int32 curProcessorId = KKU::osGetThreadId ();
188 
189  int32 errorCode = 0; // 0=No Error;
190  // 1=There is no Block
191  // 2=Different thread holds the block
192  // 3=Failure to get a lock
193 
194  {
195  CriticalSectionStart ();
196 
197  if (!blocked)
198  {
199  errorCode = 1;
200  }
201 
202  else if (curProcessorId != blockerThreadId)
203  {
204  errorCode = 2;
205  }
206 
207  else
208  {
209  blockerDepth--;
210  if (blockerDepth < 1)
211  {
212  blocked = false;
213  blockerThreadId = -1;
214  blockerDepth = 0;
215  }
216  }
217 
218  CriticalSectionEnd ();
219  }
220 
221  if (errorCode == 0)
222  return;
223 
224  else if (errorCode == 1)
225  throw KKStr ("GoalKeeper::EndBlock Name[" + name + "] There was no block established.");
226 
227  else if (errorCode == 2)
228  throw KKStr ("GoalKeeper::EndBlock Name[" + name + "] ThreadId[" + curProcessorId + "] Curenly holds Block; our ThreadId[" + curProcessorId + "]");
229 
230  return;
231 } /* EndBlock */
