283  void ConfigurationLoad ()
284  {
285  DirectoryField->Items->Clear ();

287 
288  System::IO::StreamReader^ i = nullptr;
289  try {i = gcnew System::IO::StreamReader (configFileName);} catch (Exception^) {i = nullptr;}
290  if (i == nullptr)
291  return;
292 
293  String^ nextLine = nullptr;
294 
295  while (true)
296  {
297  try {nextLine = i->ReadLine ();} catch (Exception^) {break;}
298  if (nextLine == nullptr)
299  break;
300 
301  nextLine = nextLine->Trim ();
302 
303  if ((nextLine->Length < 3) || (nextLine->Substring (0, 2) == "//"))
304  continue;
305 
306  array<String^>^ fields = nextLine->Split ('\t');
307  if (fields->Length != 2)
308  continue;
309 
310  String^ fieldName = fields[0];
311  String^ fieldValue = fields[1];
312 
313  if (fieldName == "DirectoryHistory")
314  DirectoryField->Items->Add (fieldValue);
315 
316  else if (fieldName == "LastFileFormat")
317  {
318  FeatureFileIOPtr driver = FeatureFileIO::FileFormatFromStr (PicesKKStr::SystemStringToKKStr (fieldValue));
319  if (driver)
320  FormatField->Text = PicesKKStr::KKStrToSystenStr (driver->DriverName ());
321  else
322  FormatField->Text = "";
323  }
324  }
325  i->Close ();
326 
327 
328  if (String::IsNullOrEmpty (FormatField->Text))
329  FormatField->Text = PicesKKStr::KKStrToSystenStr (FeatureFileIOPices::Driver ()->DriverName ());
