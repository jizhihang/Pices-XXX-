143 void ExtractionManager::CancelProcessing (int32 miliSecsToWait)
144 {
145  cancelFlag = true;
146  if (allThreads)
147  {
148  ImageExtractionThreadList::iterator idx;
149  for (idx = allThreads->begin (); idx != allThreads->end (); ++idx)
150  {
151  ImageExtractionThreadPtr thread = *idx;
152  thread->CancelThread ();
153  }
154  }
155 
156  if (miliSecsToWait == 0)
157  return;
158 
159  bool allThreadsStopped = true;
160  uint32 numThreadsStopped = 0;
161  uint32 numThreadsStillRunning = 0;
162 
163  int32 numMiliSecsSoFar = 0;
164 
165  while (true)
166  {
167  allThreadsStopped = true;
168  numThreadsStopped = 0;
169  numThreadsStillRunning = 0;
170  ImageExtractionThreadList::iterator idx;
171  for (idx = allThreads->begin (); idx != allThreads->end (); ++idx)
172  {
173  ImageExtractionThreadPtr thread = *idx;
174  if ((thread->Status () == tsStopped) ||
175  (thread->Status () == tsNotStarted) ||
176  (thread->Crashed ())
177  )
178  {
179  numThreadsStopped++;
180  }
181  else
182  {
183  numThreadsStillRunning++;
184  allThreadsStopped = false;
185  }
186  }
187 
188  if (allThreadsStopped)
189  {
190  break;
191  }
192  else
193  {
194  osSleep (0.01f);
195  numMiliSecsSoFar += 10;
196  }
197 
198  if ((miliSecsToWait >= 0) && (numMiliSecsSoFar > miliSecsToWait))
199  break;
200  }
201 
202  if (numThreadsStillRunning > 0)
203  {
204  ImageExtractionThreadList::iterator idx;
205  for (idx = allThreads->begin (); idx != allThreads->end (); ++idx)
206  {
207  ImageExtractionThreadPtr thread = *idx;
208  thread->Stop ();
209  }
210  }
211 } /* CancelProcessing */

918 void ExtractionManager::ManageTheExtraction (bool& successful)
919 {
920  Initialize (successful);
921  if (!successful)
922  {
923  doneExecuting = true;
924  return;
925  }
926 
927  if (CancelFlag ())
928  return;
929 
930  StartThreads (successful);
931  if (!successful)
932  {
933  CancelProcessing (5000);
934  doneExecuting = true;
935  return;
936  }
937 
938  if (!CancelFlag ())
939  {
940  // Image Extraction can not be done until all frames have been extracted; so for this reason we only
941  // monitir the 'frameExtractorThread' thread. After the 'frameExtractorThread' has processed all avaialble
942  // sipper data we will then no longer need to monitorthis thread.
943  uint32 loopCount = 0;
944  ImageExtractionThreadList frameExtractionThreads (false);
945  frameExtractionThreads.PushOnBack (frameExtractorThread);
946  MonitorUntilDone (&frameExtractionThreads, loopCount, successful);
947  if (!successful)
948  {
949  doneExecuting = true;
950  return;
951  }
952 
953  if (!CancelFlag ())
954  {
955  // Even though there is no more frames being produced by 'frameExtractorThread' we still have to make
956  // sure that all 'frames' that are on queue to be proessed are processed.
957  MonitorUntilDone (frameProcessors, loopCount, successful);
958  if (!successful)
959  {
960  doneExecuting = true;
961  return;
962  }
963 
964  if (!CancelFlag ())
965  {
966  // At this point all data has been extracted and all frames processed; we only need to make sure that all
967  // the images that have been queued for insertion into the database have been processed.
968  ImageExtractionThreadList dataBaseUpdateThreads (false);
969  dataBaseUpdateThreads.PushOnBack (dataBaseUpdaterThread);
970  MonitorUntilDone (&dataBaseUpdateThreads, loopCount, successful);
971  if (!successful)
972  {
973  doneExecuting = true;
974  return;
975  }
976  }
977  }
978  }
979 
980  if (AnyProcessorsCrashed ())
981  {
982  reportFile << endl
983  << endl
984  << "One or more Threads crashed and as a result the extraction did not complete corectly." << endl
985  << endl;
986 
987  log.Level (10) << endl
988  << "ExtractionManager::ManageTheExtraction One or more threads crashed extraction is not complete." << endl
989  << endl;
990  CancelProcessing (10000);
991  doneExecuting = true;
992  crashed = true;
993  return;
994  }
995 
996  if ((!cancelFlag) && (imageManager != NULL))
997  imageManager->Flush ();
998 
999  endTime = osGetLocalDateTime ();
1000  endCPUsecs = osGetSystemTimeUsed ();
1001 
1002  GenerateReport ();
1003 
1004  if (cancelFlag)
1005  completionStatus = "Extraction Canceled";
1006 
1007  else if (!crashed)
1008  completionStatus = "Done";
1009 
1010  double cpuTimeUsed = endCPUsecs - startCPUsecs;
1011 
1012  ReUpdateValidationData ();
1013 
1014  if ((dbConn != NULL) && (logEntry != NULL))
1015  {
1016  dbConn->LogEntriesProcessEnd (logEntry, (float)cpuTimeUsed, completionStatus);
1017  delete logEntry;
1018  logEntry = NULL;
1019  }
1020 
1021  if (dbConn)
1022  {
1023  if ((!cancelFlag) && (!crashed))
1024  dbConn->SipperFilesUpdateExtractionStatus (sipperRootName, '3');
1025  }
1026 
1027  doneExecuting = true;
1028 } /* ManageTheExtraction */
