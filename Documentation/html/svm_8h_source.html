194  int32 MemoryConsumedEstimated () const
195  {
196  int32 memoryConsumedEstimated = sizeof (svm_model)
197  + param.MemoryConsumedEstimated ()
198  + (int32)exampleNames.size () * 40;
199 
200  if (SV) memoryConsumedEstimated += sizeof (svm_node*) * l;
201  if (sv_coef) memoryConsumedEstimated += (nr_class - 1) * sizeof (double*) + l * (nr_class - 1) * sizeof (double);
202  if (rho) memoryConsumedEstimated += l * sizeof (double);
203  if (label) memoryConsumedEstimated += nr_class * sizeof (int32);
204  if (nSV) memoryConsumedEstimated += nr_class * sizeof (int32);
205  if (featureWeight) memoryConsumedEstimated += dim * sizeof (double);
206  if (kValueTable) memoryConsumedEstimated += sizeof (double) * l;
207  if ((xSpace != NULL) && weOwnXspace)
208  memoryConsumedEstimated += sizeof (svm_node) * l;
209 
210  return memoryConsumedEstimated;
