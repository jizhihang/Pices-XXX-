828  example->AddFeatureData (fieldNum, (float)code);
829  break;
830  }
831 
832 
833  case SymbolicAttribute:
834  {
835  int32 code = -1; // Initialize to value for missing data.
836  if (field == "?")
837  {
838  // Will flag this entry as having missing data.
839  example->MissingData (true);
840  }
841  else
842  {
843  // This is not a missing data.
844 
845  if (attributeTable[fieldNum]->Name ().EqualIgnoreCase ("ImageFileName"))
846  imageFileName = field;
847 
848  code = attributeTable[fieldNum]->GetNominalCode (field);
849  if (code < 0)
850  {
851  bool alreadyExists = false;
852  attributeTable[fieldNum]->AddANominalValue (field, alreadyExists);
853  code = attributeTable[fieldNum]->GetNominalCode (field);
854  }
855  }
856 
857  example->AddFeatureData (fieldNum, (float)code);
858  break;
859  }
860 
861  default:
862  _log.Level (-1) << endl << endl
863  << "FeatureFileIOC45::LoadFile *** Undefined Field Type ***" << endl
864  << endl;
865  break;
866 
867  } /* End of switch */
868 
869  field = C45ReadNextToken (_in, " ,", eof, eol);
870  }
871 
872  // 'field' should have the class name in it
873  if ((field.Empty ()) || eol)
874  {
875  _errorMessage << "Line[" << lineCount << "] Missing ClassName.";
876  _log.Level (-1) << endl << endl
877  << "FeatureFileIOC45::LoadFile " << _errorMessage << endl
878  << endl;
879 
880  delete examples; examples = NULL;
881  delete example; example = NULL;
882  return NULL;
883  }
884 
885  MLClassConstPtr mlClass = NULL;
886 
887  if (field == "?")
888  {
889  // The class is unknown
890  mlClass = _fileDesc->LookUpUnKnownMLClass ();
891  }
892  else
893  {
894  mlClass = _fileDesc->LookUpMLClassByName (field);
895  if (!mlClass)
896  {
897  lineIsValid = false;
898  _errorMessage << "Line[" << lineCount << "] Invalid Class[" << field << "]";
899  _log.Level (-1) << endl << endl
900  << "FeatureFileIOC45::LoadFile " << _errorMessage << endl
901  << endl;
902  delete examples; examples = NULL;
903  delete example; example = NULL;
904  return NULL;
905  }
906  }
907 
908  example->MLClass (mlClass);
909 
910  if (imageFileName.Empty ())
911  imageFileName = fileRootName + "_" + StrFormatInt (lineCount, "ZZZZZ0");
912 
913  example->ImageFileName (imageFileName);
914 
915 
916  if (lineIsValid)
917  examples->PushOnBack (example);
918 
919  lineCount++;
920 
921  if (!eof)
922  {
923  // Consume the rest of the characters in the line so that the next pass starts at
924  // the beginning of th enext line.
925  int32 ch = _in.peek ();
926  while ((ch != '\n') && (ch != '\r') && (!_in.eof ()))
927  {
928  _in.get ();
929  ch = _in.peek ();
930  }
931  if (!_in.eof ())
932  {
933  _in.get ();
934  if ((ch == '\n') && (_in.peek () == '\r'))
935  _in.get ();
936 
937  else if ((ch == '\r') && (_in.peek () == '\n'))
938  _in.get ();
939  }
940  }
941 
942  if ((lineCount % 1000) == 0)
943  cout << "Records Loaded " << lineCount << endl;
944 
945  if ((long)examples->size () > _maxCount)
946  break;
947  }
948 
949 
950  delete [] attributeTable;
951  return examples;
952 } /* LoadFile */

987  namesFileName = leedingPart + "names";
988  dataFileName = fileName;
989  }
990 
991  else
992  {
993  namesFileName = fileName + ".names";
994  dataFileName = fileName;
