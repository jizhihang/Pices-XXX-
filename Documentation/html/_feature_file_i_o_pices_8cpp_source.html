883 /** LoadInSubDirectoryTree
884  * @brief Loads in the contents of a sub-directory tree.
885  * Meant to work with SIPPER plankton images, it starts at a specified subdirectory and
886  * transverses all subdirectories. It makes use of FeatureDataReSink for each specific
887  * sub-directory.
888  * @param[in] rootDir, Strating directory.
889  * @param[in,out] mlClasses, List of classes, any new classes in fileName will be added.
890  * @param[in] useDirectoryNameForClassName, if true set class names to sub-directory name.
891  * This happens because the user may manually move images between dorectories using
892  * the sub-directory name as the class name.
893  * @param[in] log, where to send diagnostic messages to.
894  * @param[in] rewiteRootFeatureFile, If true rewite the feature file in the specified 'rootDir'. This
895  * feature file will contain all entries from all sub-directories below it.
896  * @return - A ImageFeaturesList container object. This object will own all the examples loaded.
897  */
898 ImageFeaturesListPtr FeatureFileIOPices::LoadInSubDirectoryTree
899  (KKStr _rootDir,
900  MLClassConstList& _mlClasses,
901  bool _useDirectoryNameForClassName,
902  volatile const bool& _cancelFlag,
903  bool _rewiteRootFeatureFile,
904  RunLog& _log
905  )
906 {
907  _log.Level (10) << "FeatureFileIOPices::LoadInSubDirectoryTree rootDir[" << _rootDir << "]." << endl;
908 
909  osAddLastSlash (_rootDir);
910 
911  KKStr featureFileName ("");
912  KKStr fullFeatureFileName ("");
913 
914  if (!_rootDir.Empty ())
915  {
916  featureFileName = osGetRootNameOfDirectory (_rootDir) + ".data";
917  fullFeatureFileName = _rootDir + featureFileName;
918  }
919  else
920  {
921  featureFileName = "Root.data";
922  fullFeatureFileName = "Root.data";
923  }
924 
925  MLClassConstPtr unKnownClass = _mlClasses.GetUnKnownClass ();
926  if (_useDirectoryNameForClassName)
927  {
928  KKStr className = MLClass::GetClassNameFromDirName (_rootDir);
929  unKnownClass = _mlClasses.GetMLClassPtr (className);
930  }
931 
932  bool changesMade = false;
933

937

939  {
940  DateTime timeStamp;

942  featureFileName,
943  unKnownClass,
944  _useDirectoryNameForClassName,
945  _mlClasses,
946  _cancelFlag,
947  changesMade,
948  timeStamp,
949  _log
950  );
951  if (_useDirectoryNameForClassName)
952  {
953  ImageFeaturesList::iterator idx;
954  for (idx = dirImages->begin (); idx != dirImages->end (); idx++)
955  {
956  if ((*idx)->MLClass () != unKnownClass)
957  {
958  (*idx)->MLClass (unKnownClass);
959  changesMade = true;
960  }
961  }
962

964  {

975  _log

987  // lets process any sub-directories.
988

990

1000  {
1001  // We ignore this director
1002  continue;
1003  }
1004

1006 
1007  ImageFeaturesListPtr subDirImages = LoadInSubDirectoryTree (newDirPath,
1008  _mlClasses,
1009  _useDirectoryNameForClassName,
1010  _cancelFlag,
1011  true, // true = ReWriteRootFeatureFile
1012  _log
1013  );
1014  ImageFeaturesPtr image = NULL;
1015 
1016  osAddLastSlash (subDirName);
1017 
1018  // We want to add the directory path to the ImageFileName so that we can later locate the source image.
1019  ImageFeaturesList::iterator idx = subDirImages->begin ();
1020  for (idx = subDirImages->begin (); idx != subDirImages->end (); idx++)
1021  {
1022  image = *idx;
1023  KKStr newImageFileName = subDirName + image->ImageFileName ();
1024  image->ImageFileName (newImageFileName);
1025  }
1026

1028  subDirImages->Owner (false);
1029  delete subDirImages;
1030  }
1031

1033  }

1177  {

1180  (imageFileName->StrInStr ("After."))
