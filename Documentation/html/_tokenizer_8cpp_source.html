292 {
293  if (firstChar == strDelChar)

296  KKStr str (20);
297 
298  // Scan until we hit another '"' character, or end of KKStr.
299  while (!atEndOfFile)

301  if (firstChar == strDelChar)
302  {
303  // We reached the end of the string
304  GetNextChar ();
305  break;
306  }

316  case 't': str.Append ('\t'); break;
