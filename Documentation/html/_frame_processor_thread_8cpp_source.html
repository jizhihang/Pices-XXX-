305 LogicalFramePtr FrameProcessorThread::GetNextFrameToProcess ()
306 {
307  LogicalFramePtr logicalFrame = framePool->GetNextFrameToProcess ();
308  while ((logicalFrame == NULL) && (!CancelFlag ()))
309  {
310  osSleep (0.01f);

312  if ((logicalFrame == NULL) && TerminateFlag ())
313  {
314  // There are no more frames needing processing and the Extractionmanager wants to stop when convienient.

323  }
