439  PicesClass AssignVallidatedClassAsPredictedClass (PicesFeatureVector fv)
440  {
441  PicesClass validatedClass = fv.MLClass;
442  if ((validatedClass == null) || (classesInClassifier == null))
443  return validatedClass;
444 
445  if (classesInClassifier.LookUpIndex (validatedClass) >= 0)
446  return validatedClass;
447 
448  // Lets see if the Validated Class is a decendent of one of the classes in the Classifier.
449  PicesClass candidate = null;
