94 void KKThreadManager::ShutdownProcessing (int32 miliSecsToWait)
95 {
96  shutdownFlag = true;
97 
98  if (!threads)
99  return;
100

102  shutdownFlag = true;
103 
104  int32 lastNumThreadsRunning = 0;
105  int32 lastNumThreadsStopped = 0;
106 
107  bool allTreadsAreShutdown = false;
108 
109  while (!allTreadsAreShutdown)
110  {
111  allTreadsAreShutdown = true;
112  int32 numThreadsRunning = 0;
113  int32 numThreadsStopped = 0;
114 
115  KKThreadList::iterator idx;
116  for (idx = threads->begin (); idx != threads->end (); ++idx)
117  {
118  KKThreadPtr thread = *idx;
119 
120  switch (thread->Status ())
121  {
122  case KKThread::tsNotStarted:
123  case KKThread::tsStopped:
124  // Nothing to do; this thread is not running,
125  ++numThreadsStopped;
126  break;
127 
128  case KKThread::tsStopping:
129  allTreadsAreShutdown = false;
130  ++numThreadsRunning;
131  break;
132 
133  case KKThread::tsRunning:
134  {

136  if (!(thread->ShutdownFlag ()))
137  {
138  if (thread->OkToShutdown ())
139  {
140  thread->ShutdownThread ();
141  numMiliSecsWaited = 0;
142  }
143  }
144  ++numThreadsRunning;
145  }
146  break;
147 
148  case KKThread::tsStarting:
149  // Have to let it finish starting before we can flag it to shutdown.
150  ++numThreadsRunning;
151  break;
152  } /* End of Switch */
153  }
154 
155  if ((numThreadsRunning != lastNumThreadsRunning) ||
156  (numThreadsStopped != lastNumThreadsStopped)
157  )
158  {
159  // The status of one or more threads changed; can reset the 'numMiliSecsWaited' clock.
160  numMiliSecsWaited = 0;
161  }
162 
163  if (!allTreadsAreShutdown)
164  {
165  if ((miliSecsToWait > 0) && (numMiliSecsWaited > miliSecsToWait))
166  {
167  // The shutdown process is taking longer than the the 'miliSecsToWait' parameter will allow for; we need to terminate
168  // all running threads now.
169  KillAllRunningThreads ();
170  break;
171  }
172  else
173  {
174  osSleep (0.01f);

176  }
177  }
178  }
