13 /*!
14  \class SipperDiskUtilities.SipperDisk
15  \brief SipperDisk is used for reading and writting to a SIPPER disk drive.
16  \details The SIPPER disk has a very simple format who's details are described below.
17  You read the Master Boot record and get the 1st sector from "Partition Entry 1".
18  The SIPPER Partition will start with one Partition decriptive block followed by
19  2048 File Control Block, on efor each SIPPER file created. The File control blocks
20  will be followed by the actual SIPPER data. SIPPER records data sequentially
21  starting at the begiining of the drive until the end of the drive. There will be
22  not fragmentation issues to deal with. Each SIPPER file will have a start and end
23  sector that is specified in its corresponding File Control Block.\n
24  \see SipperDiskUtilities.PartitionTable
25  \see SipperDiskUtilities.SipperFileControlBlock
26  \see SipperDiskUtilities.SipperDiskFile
27  *
28  \code
29  *******************************************************************************************
30  * Sipper Disk Drive Layout
31  *
32  * Master Boot Record
33  * |Boot-Record-Code (0-445)
34  * |
35  * |Partition-Table
36  * | Partition-Entry-0 (446 - 461)
37  * | Boot-Id (446)
38  * | Start-Head (447)
39  * | Start-Sector (448)
40  * | Start-Cylinder (449)
41  * | System-ID (450)
42  * | EndHead (451)
43  * | EndSector (452)
44  * | End-Cylinder (453)
45  * | Relative-Sector (454 - 457)
46  * | totalSectors (458 - 461)
47  * |
48  * | Partition-Entry-1 (462 - 477) // Sipper will be partition "1".
49  * | Partition-Entry-2 (478 - 493)
50  * | Partition-Entry-3 (494 - 509)
51  * |
52  * |End-Of-Sector (510 - 511)
53  *
54  *
55  *
56  * (**********************************************************)
57  * (* Start of Sipper Partition *)
58  * (* The first sector will be the 'SipperPartitionSector' *)
59  * (**********************************************************)
60  *
61  * Sipper-Partition-Sector Starts at ('Partition-Entry-1'->'Relative-Sector') * 'Logical-Block-Size(512)'
62  * | Sipper-Id ( 0 - 11)
63  * | Total-Sectors (12 - 15)
64  * | File-Data-Sector (16 - 19)
65  * | End-Sector (20 - 23)
66  * | Next-Fcb-Sector (24 - 27)
67  * | Next-File-Handle (28 - 31)
68  * | Unused-Space (32 - 511)
69  *
70  *
71  *
72  * (**********************************************************)
73  * (* followed by 2048 'SipperFileControlBlock' sectors. *)
74  * (* where each one represents a single Sipper File. *)
75  * (**********************************************************)
76  *
77  * Sipper-File-Control-Block
78  * | Allocated ( 0 - 3)
79  * | Status ( 4 - 7) 0=Closed, 1=Opened
80  * | Start-Sector ( 8 - 11)
81  * | End-Sector (12 - 15)
82  * | Current-Access-Sector (16 - 19)
83  * | File-Name (20 - 49)
84  * |
85  * | filler (50 - 51) For some reason SIPPER is inserting 2 empty bytes here
86  * | probably beciause it is accessing memory on 4-byte boundaries.
87  * |
88  * | Date-Time-Stamp (52 - 59)
89  * | Second (52)
90  * | Minute (53)
91  * | Hour (54)
92  * | Unused (55)
93  * | Day (56)
94  * | Date (57) (1 -> 31)
95  * | Month (58) (1 -> 12)
96  * | Year (59) (0 -> 99) ex: 7 = 2007.
97  * |
98  * | Description (60 - 364)
99  * |
100  * | UnUsed (365 - 511)
101  * // Unused Space
102  \endcode
103  */

845  // For now we will always return true;
846  return true;

849  BlockStart ();
850 
851  long lastFsPosition = 0;
852 
853  bool diskFileWasAlreadyOpened = false;
854  if (fs != null)

857  lastFsPosition = fs.Position;
858  }
