385  if ((gpsData == null) || cancelRequested)
386  return;

388  // Filter out noisey GPS data.
389 
390  PicesGPSDataPointList fliteredGPSData = new PicesGPSDataPointList ();
391  double lastLat = 0.0;
392  double lastLong = 0.0;
393 
394  foreach (PicesGPSDataPoint dp in gpsData)
395  {
396  if ((Math.Abs (dp.AvgLongitude ) < 0.1) || (Math.Abs (dp.AvgLatitude) < 0.1))
397  continue;
398  if ((dp.AvgLatitude == lastLat) && (dp.AvgLongitude == lastLong))
399  continue;
400  fliteredGPSData.Add (dp);
401  lastLong = dp.AvgLongitude;
402  lastLat = dp.AvgLatitude;
403  }
404 
405  gpsData = fliteredGPSData;
406  fliteredGPSData = null;

408  if (gpsData.Count < 2)
409  {
410  fliteredGPSData = gpsData;
411  }
412  else
413  {
414  double totalSquareDist = 0.0;
415  double[] distBP = new double[gpsData.Count - 1];
416  double totalSquareDistSquare = 0.0;
417  for (int x = 0; x < (gpsData.Count - 1); ++x)
418  {
419  double deltaLong = gpsData[x + 1].AvgLongitude - gpsData[x].AvgLongitude;
420  double deltaLat = gpsData[x + 1].AvgLatitude - gpsData[x].AvgLatitude;
421  double squareDist = deltaLong * deltaLong + deltaLat * deltaLat;
422  totalSquareDist += squareDist;
423  totalSquareDistSquare += squareDist * squareDist;
424  distBP[x] = squareDist;
425  }
426 
427  double meanSquareDist = totalSquareDist / distBP.Length;
428  double variance = (totalSquareDistSquare / distBP.Length) - (meanSquareDist * meanSquareDist);
429  double stdDev = Math.Sqrt (variance);
430 
431  double distSquareTH = stdDev* 10;
432  int firstIdx = 0;
433  if (distBP.Length > 11)
434  {
435  for (int x = 0; x < 10; ++x)
436  {
437  if (distBP[x] > distSquareTH)
438  firstIdx = x + 1;
439  }
440  }
441 
442  fliteredGPSData = new PicesGPSDataPointList ();
443  for (int x = firstIdx; x < gpsData.Count; ++x)

956  private void ProfileChart_MouseClick (object sender, MouseEventArgs e)
957  {
958  Point p = e.Location;
959  //MouseClickLocation.Text = p.X.ToString () + "," + p.Y.ToString ();
960  ChartArea ca = ProfileChart.ChartAreas[0];
961  double longitude = ca.AxisX.PixelPositionToValue ((double)p.X);
962  double latitude = ca.AxisY.PixelPositionToValue ((double)p.Y);
963 
964  // Determine Threshold.
965  int thInPixels = 25;
966  double x = ca.AxisX.PixelPositionToValue ((double)(p.X + thInPixels));
967  double degLongPerPixel = Math.Abs (x - longitude) / thInPixels;
968 
969  double y = ca.AxisY.PixelPositionToValue ((double)(p.Y + thInPixels));
970  double degLatPerPixel = Math.Abs (y - latitude) / thInPixels;
971 
972  double longTH = degLongPerPixel * thInPixels;
973  double latTH = degLatPerPixel * thInPixels;
974 
975  PicesGPSDataPoint closestPoint = null;
976  int closestPointIndex = -1;
977  int closestPointSeriesIndex = -1;
978  int seriesIndex = 0;
979  double closestPointDistSquare = double.MaxValue;
980  foreach (DataSeriesToPlot dstp in series)
981  {
982  if (dstp.WithInThreshold (longitude, longTH, latitude, latTH))
983  {
984  double distSquare = double.MaxValue;
985  int closestPointInSeriesIndex = -1;
986  PicesGPSDataPoint closestPointInSeries = dstp.ClosestPoint (longitude, longTH, latitude, latTH, ref closestPointInSeriesIndex, ref distSquare);
987  if (closestPointInSeries != null)
988  {
989  if (distSquare < closestPointDistSquare)
990  {
991  closestPointIndex = closestPointInSeriesIndex;
992  closestPointSeriesIndex = seriesIndex;
993  closestPointDistSquare = distSquare;
994  closestPoint = closestPointInSeries;
995  }
996  }
997  }
998  seriesIndex++;
999  }
1000 
1001  if (closestPoint != null)
1002  HighLightClosestPoint (closestPointSeriesIndex, closestPointIndex, closestPoint);
1003 
1004  String gpsStr = PicesMethods.LatitudeLongitudeToString (latitude, longitude);
1005 
1006  CurGPSLocation.Text = gpsStr;
1007  }
