142  private void ValidateDeploymentNum()
143  {
144  errorProvider1.SetError (DeploymentNum, null);
145  if (!addingNewDeployment)
146  return;
147 
148  String s = DeploymentNum.Text.Trim ();

152  validationErrorsFound = true;
153  return;

155 
156  String invalidCharacters = "";
157  for (int x = 0; x < s.Length; x++)
158  {
159  char ch = char.ToUpper (s[x]);
160  bool validChar = ((ch >= '0') && (ch <= '9')) || ((ch >= 'A') && (ch <= 'Z'));
161  if (!validChar)
162  {
163  if (invalidCharacters.Length > 0)
164  invalidCharacters += ", ";
165  invalidCharacters += s[x].ToString ();
166  }
167  }
168 
169  if (invalidCharacters.Length > 0)
170  {
171  errorProvider1.SetError (DeploymentNum, "Invalid characters[" + invalidCharacters + "]");
172  validationErrorsFound = true;
173  return;
174  }
175 
176  if (s.Length > 4)
177  {
178  errorProvider1.SetError (DeploymentNum, "DeploymentNum[" + s + "] is to long; maximum of 4 characters allowed.");
179  validationErrorsFound = true;
180  return;
181  }
182 
183  if ((addingNewDeployment) || (s != deployment.DeploymentNum))
184  {
185  PicesSipperDeployment existingDeployment = DbConn ().SipperDeploymentLoad (CruiseName.Text, StationName.Text, s);
186  if (existingDeployment != null)
187  {
188  errorProvider1.SetError (DeploymentNum, "DeploymentNum[" + s + "] is already been used for Cruise[" + CruiseName.Text + "], Station[" + StationName.Text + "].");
189  validationErrorsFound = true;
190  return;
191  }
192  }
193 
194  DeploymentNum.Text = s;
