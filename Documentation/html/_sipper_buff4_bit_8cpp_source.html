365 void SipperBuff4Bit::BuildConversionTables ()
366 {
367  GlobalGoalKeeper::StartBlock ();
368  if (!convTable4BitTo8Bit)
369  {
370  int32 x = 0;
371  int32 y = 0;
372  int32 inc = 256 / 16;

374  convTable8BitTo4Bit = new uchar[256];
375  for (x = 0; x < 16; ++x)

377  int32 this8Bit = x * inc;
378  int32 next8Bit = (x + 1) * inc;
379

383  for (y = this8Bit; y < next8Bit; ++y)
384  convTable8BitTo4Bit[y] = x;

553  bool eol = false;

555  OpRec rec;
556  OpRec rec2;
557  uint32 recsRead = 0;
558 
559  uint32 bufferLineLen = 0;
560 
561  uchar* textMsg = NULL;
562  uint32 textMsgLen = 0;
563 
564  lineLen = 0;
565 
566  do
567  {
568  recsRead = fread (&rec, sizeof (rec), 1, inFile);
569  if (recsRead == 0)
570  {
571  break;
572  }
573 
574  opCode = rec.textBlock1.opCode;
575 
576  if (opCode == 0)
577  eol = true;
578 
579  else if (bufferLineLen >= lineBuffSize)
580  {
581  // Something has gone wrong, we should have encountered a eol opCode before this point; that is the
582  // length of the encoded line is exceeding the scan line length.
583  eol = true;
584  ungetc (rec.textChar, inFile);
585  break;
586  }
587 
588  else if (opCode == 1)
589  ProcessTextBlock (rec);
590 
591  else if (opCode == 2)
592  ProcessInstrumentDataWord (rec);
593 
594  else if ((opCode >= 4) && (opCode <= 9))
595  {
596  // OpCode determines RunLen (4=2, 5=3, ... 9=7)
597  uint32 runLen = opCode - 2;
598  runLenChar = convTable4BitTo8Bit [rec.runLen.pixelValue];
599  uint32 newLineSize = bufferLineLen + runLen;
600 
601  if (newLineSize > lineBuffSize)
602  {
603  cerr << "SipperBuff4Bit::GetNextScanLine ***ERROR*** Exceeding 'bufferLineLen'; ScanLine[" << curRow << "]." << endl;
604  cerr << " newLineSize: " << newLineSize << endl;
605  }
606  else
607  {
608  memset (&(lineBuff[bufferLineLen]), runLenChar, runLen);
609  bufferLineLen = newLineSize;
610  }
611  }
612 
613  else if (opCode == 10) /* OpRecRun256Len1 */
614  {
615  // Run-Length (1 thru 256 Pixels).
616  recsRead = fread (&rec2, sizeof (rec2), 1, inFile);
617  if (recsRead < 1)
618  eol = true;
619  else
620  {
621  uint32 runLen = 1 + rec2.run256Len2.runLen;
622  uint32 newLineSize = bufferLineLen + runLen;
623  if (newLineSize > lineBuffSize)
624  {
625  cerr << "SipperBuff4Bit::GetNextScanLine ***ERROR*** Exceeding 'bufferLineLen'; ScanLine[" << curRow << "]." << endl;
626  }
627  else
628  {
629  uchar runLenChar = convTable4BitTo8Bit [rec.run256Len1.pixelValue];
630  memset (&(lineBuff[bufferLineLen]), runLenChar, runLen);
631  bufferLineLen = newLineSize;
632  }
633  }
634  }
635 
636  else if (opCode == 11)
637  {
638  // Raw-String (1 Pixel Length).
639  if (bufferLineLen < lineBuffSize)
640  {
641  lineBuff[bufferLineLen] = convTable4BitTo8Bit[rec.raw1Pixel.pixelValue];
642  ++bufferLineLen;
643  }
644  else
645  {
646  cerr << "SipperBuff4Bit::GetNextScanLine ***ERROR*** Exceeding 'bufferLineLen'; ScanLine[" << curRow << "]." << endl;
647  }
648  }
649 
650  else if (opCode == 12)
651  {
652  // Raw-String (Even length 2 thru 32).
653  uint16 numRawRecs = rec.raw32Pixels.len + 1; // We add 1 to 'numRawRecs' because '1' was subtracted out when written to Scanner File.
654  uint16 numRawPixels = numRawRecs * 2;
655  uint16 newBufferLineLen = bufferLineLen + numRawPixels;
656  if (newBufferLineLen > lineBuffSize)
657  {
658  cerr << "SipperBuff4Bit::GetNextScanLine ***ERROR*** Exceeding 'bufferLineLen'; ScanLine[" << curRow << "]." << endl;
659  }
660  ProcessRawPixelRecs (numRawRecs, lineBuff, lineBuffSize, bufferLineLen);
661  }
662 
663  else if (opCode == 13)
664  {
665  // Raw-String (Odd Length 1 thru 513).
666  recsRead = fread (&rec2, sizeof (rec2), 1, inFile);
667  if (recsRead < 1)
668  {
669  eol = true;
670  eof = true;
671  }
672  else
673  {
674  uint16 numRawRecs = 1 + 16 * (uint16)(rec.raw513Pixels1.lenHigh) + (uint16)(rec2.raw513Pixels2.lenLow);
675  uint16 numRawPixels = 1 + 2 * numRawRecs;
676  uint16 newBufferLineLen = bufferLineLen + numRawPixels;
677  if (newBufferLineLen > lineBuffSize)
678  {
679  cerr << "SipperBuff4Bit::GetNextScanLine ***ERROR*** Exceeding 'bufferLineLen'; ScanLine[" << curRow << "]." << endl;
680  }
681 
682  if (bufferLineLen < lineBuffSize)
683  {
684  lineBuff[bufferLineLen] = convTable4BitTo8Bit[rec2.raw513Pixels2.pix0];
685  ++bufferLineLen;
686  }
687 
688  ProcessRawPixelRecs (numRawRecs, lineBuff, lineBuffSize, bufferLineLen);
689  }
690  }
691  } while (!eol);
692 
693  lineLen = bufferLineLen;
694 
695  ++curRow;
696 
697  return;
698 } /* GetNextScanLine */
